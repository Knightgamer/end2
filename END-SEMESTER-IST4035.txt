  <script>
    function randomize() {
      var divfruits = document.getElementById("container");
      var elementsArray = Array.prototype.slice.call(
        divfruits.getElementsByClassName("randomize")
      );
      elementsArray.forEach(function (element) {
        divfruits.removeChild(element);
      });
      shuffleArray(elementsArray);
      elementsArray.forEach(function (element) {
        divfruits.appendChild(element);
      });
    }

    function shuffleArray(array) {
      for (var i = array.length - 1; i > 0; i--) {
        var j = Math.floor(Math.random() * (i + 1));
        var temp = array[i];
        array[i] = array[j];
        array[j] = temp;
      }
      return array;
    }

    function arrange() {
      var elements = document.getElementsByTagName("div");
      var array = [];
      for (var i = 0; i < elements.length; i++) {
        array.push(elements[i].innerHTML);
      }
      array.sort();
      for (var i = 0; i < elements.length; i++) {
        elements[i].innerHTML = array[i];
      }
    }
